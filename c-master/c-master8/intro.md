# ファイル操作プログラムの概要

#include <stdio.h> ← FILE 型の定義が入っている

main() {

    FILE *fp;

    fp = fopen("ファイル名", "モード"); ← FILE型構造体を生成(ファイルをオープンする)

    ********************************

    fpを通してFILE型を扱う操作関数で
    ファイルを入出力

    ********************************

    fclose(fp); ← FILE型構造体の解放(ファイルをクローズする)

}
